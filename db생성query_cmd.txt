create database software_engineering;
use software_engineering;
create table store_info(
storeID int(5) unsigned not null auto_increment primary key,
storeName varchar(20) not null,
storeLocation varchar(50) not null,
tableLockList JSON,
maxSpace tinyint unsigned NOT NULL default 0,
emptyColor tinyint unsigned not null default 255
);
create table store_status(
storeID int(5) unsigned not null,
open bool default false not null,
currentStatus tinyint unsigned NOT NULL default 0
);
alter table store_status add constraint rule1 foreign key (storeID)
references store_info (storeID) ON UPDATE CASCADE;
alter table store_status add constraint rule2 foreign key (storeID)
references store_info (storeID) ON DELETE CASCADE;
desc store_status;
desc store_info;
